# -*- coding: utf-8 -*-


from dabo import ui
from dabo.base_object import dObject
from dabo import events
from dabo.lib import datanav as datanav

IGNORE_STRING = datanav.QRY_OPERATOR.IGNORE


# Controls for the select page:
class SelectControlMixin(dObject):
    def initProperties(self):
        super().initProperties()
        self.SaveRestoreValue = True

class SelectTextBox(SelectControlMixin, ui.dTextBox): pass
class SelectCheckBox(SelectControlMixin, ui.dCheckBox): pass
class SelectLabel(SelectControlMixin, ui.dLabel):
    def afterInit(self):
        # Basically, we don't want anything to display, but it's
        # easier if every selector has a matching control.
        self.Caption = ""
class SelectDateTextBox(SelectControlMixin, ui.dDateTextBox): pass
class SelectSpinner(SelectControlMixin, ui.dSpinner): pass

class SelectionOpDropdown(ui.dDropdownList):
    def initProperties(self):
        super().initProperties()
        self.SaveRestoreValue = True
        self.ValueMode = "String"

    def afterInit(self):
        super().afterInit()
        # Ensure we always have a sane default operator selected.
        if not self.StringValue:
            try:
                self.StringValue = IGNORE_STRING
            except Exception:
                pass

    def initEvents(self):
        super().initEvents()
        self.bindEvent(events.Hit, self.onChoiceMade)
        self.bindEvent(events.ValueChanged, self.onValueChanged)

    def onValueChanged(self, evt):
        # italicize if we are ignoring the field:
        val = self.StringValue or ""
        self.FontItalic = (IGNORE_STRING in val)
        if self.Target:
            self.Target.FontItalic = self.FontItalic

    def onChoiceMade(self, evt):
        val = self.StringValue
        if val and IGNORE_STRING not in val and self.Target:
            # A comparison op was selected; let 'em enter a value
            self.Target.setFocus()

    def _getTarget(self):
        try:
            _target = self._target
        except AttributeError:
            _target = self._target = None
        return _target

    def _setTarget(self, tgt):
        self._target = tgt
        if self.Target:
            self.Target.FontItalic = self.FontItalic

    Target = property(_getTarget, _setTarget, None, "Holds a reference to the edit control.")


class SortLabel(ui.dLabel):
    def initEvents(self):
        super().initEvents()
        self.bindEvent(events.MouseRightClick, self.Parent.Parent.onSortLabelRClick)
        # Add a property for the related field
        self.relatedDataField = ""


class PagSelectBase(datanav.SelectPage):
